# Energy Optimization for Mountain Terrain
# Calculates battery drain based on ascent/descent

def calculate_energy(path):
    total_energy = 0
    for i in range(len(path)-1):
        p1, p2 = path[i], path[i+1]
        dist = get_dist(p1, p2)
        climb = p2.z - p1.z
        
        # Gravity Penalty Factor
        penalty = 1.0
        if climb > 0: penalty = 2.5 
        
        total_energy += dist * penalty * BURN_RATE
        
    return total_energy
